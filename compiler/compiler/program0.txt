PROGRAM IS
a, b, c, d, e, f, g, h, tab[1:5]
BEGIN
    # Przypisanie wartości do zmiennych i tablicy
    a := -2;
    b := 0;
    tab[1] := 10;
    tab[2] := -53498223;
    tab[3] := 20;
    tab[4] := 0;
    tab[5] := -1;

    # 1. Mnożenie przez 0
    IF b = 0 THEN
        c := tab[1] * b; # Mnożenie przez 0
        WRITE c; # Powinno wypisać 0
    ENDIF

    # 2. Mnożenie dwóch wartości ujemnych
    IF tab[2] < 0 THEN
        d := tab[2] * tab[5]; # Mnożenie dwóch ujemnych wartości
        WRITE d; # Powinno wypisać 15
    ENDIF

    # 3. Mnożenie wartości dodatniej przez ujemną
    IF tab[3] > 0 THEN
        e := tab[3] * tab[2]; # Mnożenie dodatniej przez ujemną
        WRITE e; # Powinno wypisać -100
    ENDIF

    # 4. Mnożenie wartości ujemnej przez 0
    IF tab[5] < 0 THEN
        f := tab[5] * tab[4]; # Mnożenie ujemnej wartości przez 0
        WRITE f; # Powinno wypisać 0
    ENDIF

    # 5. Mnożenie zmiennej przez ujemny element tablicy
    IF a < 0 THEN
        g := a * tab[2]; # Mnożenie zmiennej przez tab[2]
        WRITE g; # Powinno wypisać 10
    ENDIF

    # 6. Mnożenie elementu tablicy przez 0
    IF tab[4] = 0 THEN
        h := tab[4] * tab[3]; # Mnożenie tab[4] przez tab[3]
        WRITE h; # Powinno wypisać 0
    ENDIF
END
