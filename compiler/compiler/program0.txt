# Silnia+Fibonacci
# ? 20
# > 2432902008176640000
# > 6765

PROGRAM IS
    f[0:100], s[0:100], i[0:100], n, k, l,p,tab[-4:5]
BEGIN
    READ n;
    p:=3;
    tab[3]:=20;
    f[0] := 0;
    s[0] := 1;
    i[0] := 0;
    f[1] := 1;
    s[1] := 1;
    i[1] := 1;
    FOR j FROM tab[p] DOWNTO n DO
        k := j - 1;
        l := k - 1;
	    i[j] := i[k] + 1;
	    f[j] := f[k] + f[l];
        s[j] := s[k] * i[j];
        WRITE j;

    ENDFOR

    WRITE s[20];
    WRITE f[20];
END
